3、B1选股

{ 综合选股公式：基于J值、成交量、趋势强弱等条件筛选 }
{ 公式说明：前2日J值在(13,55)，当日J值<13；量能递减（当日<前2日且昨日<前2日）；白线>黄线；收盘>黄线97% }

{ 参数区}
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=20; SS:=3; { 斜率周期与斜率平滑周期 }
J_LOW:=13; J_HIGH:=60; YELLOW_RATIO:=0.95; { J阈值与黄线比例阈值 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 4) > J_HIGH AND REF(J, 4) > J;       { 前5日J值严格递减，且当日J小于J_LOW }
条件2 := J < J_LOW;                                   { 当日J值小于J_LOW }
条件3 := VOL < REF(VOL, 2) AND VOL < REF(VOL, 1);                        { 当日成交量小于前2日}
条件4 := ZX_WHITE > ZX_YELLOW;                     { 白线大于黄线 }
条件5 := CLOSE > ZX_YELLOW * YELLOW_RATIO;                 { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }
SLOPE_W_RAW := (ZX_WHITE - REF(ZX_WHITE, SP)) / SP;
SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;
SLOPE_W := EMA(SLOPE_W_RAW, SS);
SLOPE_Y := EMA(SLOPE_Y_RAW, SS);
条件6 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 均量递减：MA(VOL,7) > MA(VOL,5) > MA(VOL,3) > VOL }
条件7 := MA(VOL,7) > MA(VOL,5) AND MA(VOL,5) > MA(VOL,3) AND MA(VOL,3) > VOL;

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7;


4、B2选股
{ 综合选股公式：基于J值、涨幅、成交量等多条件筛选 }
{ 公式说明：前日J值<13，当日J值<55，涨幅>3.5%，收阳线，放量，股价在5日均线上方 }

{ 参数区 }
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=30; SS:=3; { 斜率周期与斜率平滑 }
J_LOW:=13; J_HIGH:=55; YELLOW_RATIO:=0.95; { J阈值与黄线比例 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 2) < J_LOW;                          { 前日J值小于J_LOW }
条件2 := J < J_HIGH;                                  { 当日J值小于J_HIGH }
条件3 := (CLOSE - REF(CLOSE, 1)) / REF(CLOSE, 1) * 100 > 3.95; { 当日涨幅大于3.95% }
条件4 := CLOSE > OPEN;                            { 当日收盘价大于开盘价 }
条件5 := VOL > REF(VOL, 1);                       { 成交量大于昨日成交量 }
条件6 := CLOSE > MA(CLOSE, 5);                    { 股价（白线）在5日均线（黄线）上方 }
条件7 := ZX_WHITE > ZX_YELLOW;
条件8 := CLOSE > ZX_YELLOW*YELLOW_RATIO;                  { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }
SLOPE_W_RAW := (ZX_WHITE - REF(ZX_WHITE, SP)) / SP;
SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;
SLOPE_W := EMA(SLOPE_W_RAW, SS);
SLOPE_Y := EMA(SLOPE_Y_RAW, SS);
条件9 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 放量：当日成交量 > 过去5日均量的 1.8 倍 }
条件10 := VOL > 1.8 * MA(VOL, 5);

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7 AND 条件8 AND 条件9 AND 条件10;

CG:=IF(MA(CLOSE,17)<FS,FS,MA(CLOSE,17));

{ 浮筹线：成本线减去3日EMA与成本线的差值 }
浮筹:=CG-(EMA(C,3)-CG);

{ 力度：趋势力度减去浮筹 }
力度:=趋势力度-浮筹;

{ --------散筹分析------------ }
{ 年线：250日EMA，代表长期趋势 }
年线:=EMA(C,250);

{ 散筹线：年线减去3日EMA与年线的差值 }
散筹:=年线-(EMA(C,3)-年线);

{ 操作线系统 }
{ 7日EMA操作线，绿色显示 }
操作线:=EMA(C,7);
A:=(REF(操作线,1));

{ ----------效率比计算-------------- }
{ 30日价格变化幅度 }
DIR2:=ABS(CLOSE-REF(CLOSE,30));

{ 30日累计价格变化 }
VIR2:=SUM(ABS(CLOSE-REF(CLOSE,1)),30);

{ 效率比：方向变化/累计变化 }
ER2:=DIR2/VIR2;

{ 平滑系数计算 }
CS2:=ER2*(0.8-2/24)+2/24;
CQ2:=CS2*CS2;

{ ----------机构评分系统（满分80分）---------- }
{ 短期均线多头排列：5日线上穿10日线得20分 }
AAC1:=IF(MA(C,5)>MA(C,10),20,0);

{ 中期均线多头排列：20日线上穿60日线得10分 }
AAC2:=IF(MA(C,20)>MA(C,60),10,0);

{ KDJ金叉：J线上穿K线得10分 }
AAC3:=IF(KDJ.J>KDJ.K,10,0);

{ MACD金叉：DIF线上穿DEA线得10分 }
AAC4:=IF(MACD.DIF>MACD.DEA,10,0);

{ MACD柱状图为正：MACD>0得10分 }
AAC5:=IF(MACD.MACD>0,10,0);

{ 成交量放大：当日成交量大于60日均量得10分 }
AAC6:=IF(V>MA(V,60),10,0);

{ 获利盘充足：获利盘大于50%得10分 }
AAC7:=IF(WINNER(C)>0.5,10,0);

{ 涨幅较大：当日涨幅大于3%得10分 }
AAC8:=IF(C/REF(C,1)>1.03,10,0);

{ 机构评分计算 }
机构评分:=AAC1+AAC2+AAC3+AAC4+AAC5+AAC6+AAC7+AAC8;
机构评分条件:=机构评分>50;
原始评分信号:=机构评分-REF(机构评分, 1) >= 50 AND REF(机构评分, 2) <= 30;

{ ---------钱多多信号系统----------- }
{ 加权平均价：3倍收盘价+最高+最低+开盘 }
JJ:=(3*C+H+L+O)/6;

{ 加权均价线：8日加权平均，权重递减 }
ABC11:=(8*JJ+7*REF(JJ,1)+6*REF(JJ,2)+5*REF(JJ,3)+4*REF(JJ,4)+3*REF(JJ,5)+2*REF(JJ,6)+REF(JJ,8))/36;

{ 钱多多1信号：放量突破信号 }
{ 条件：成交量创10日新高 AND 成交量大于前日2倍 AND 收盘价高于加权均价 AND 收阳线 }
钱多多1:=VOL=HHV(VOL,10) AND VOL>2*REF(VOL,1) AND CLOSE>ABC11 AND C>REF(C,1);

{ 过滤信号：5日内只显示一次 }
LJL:=FILTER(钱多多1,5);

{ 钱多多2信号：突破前期高点确认 }
{ 钱多多1信号后1-3日内，收盘价突破前期高点 }
钱多多2:=IF(BARSLAST(LJL)=1 AND CLOSE>REF(HIGH,1),1,IF(BARSLAST(LJL)=2 AND CLOSE>REF(HIGH,2),1,IF(BARSLAST(LJL)=3 AND CLOSE>REF(HIGH,3),1,0)));
TP:=钱多多2=1;

{ 钱多多3信号：突破前期低点确认 }
{ 钱多多1信号后1-3日内，收盘价突破前期低点 }
钱多多3:=IF(BARSLAST(LJL)=1 AND CLOSE>REF(LOW,1),1,IF(BARSLAST(LJL)=2 AND CLOSE>REF(LOW,2),1,IF(BARSLAST(LJL)=3 AND CLOSE>REF(LOW,3),1,0)));
STFZ:=钱多多3=1;

{ 综合信号：同时满足钱多多2和钱多多3 }
CL:=STFZ AND TP;

{ 过滤综合信号：3日内只显示一次 }
钱多多CL:=FILTER(CL,3);

{ ----------短期多空线底部拐点信号系统---------- }

{ 计算短期多空线的线性拟合斜率 }
短期多空线斜率:=SLOPE(短期多空线, 5);                    { 5日线性拟合斜率 }
前日短期斜率:=REF(短期多空线斜率, 1);                     { 前一日斜率 }
前2日短期斜率:=REF(短期多空线斜率, 2);                    { 前2日斜率 }

{ 斜率变化趋势判断 }
斜率上升:=短期多空线斜率 > 前日短期斜率;                  { 当前斜率大于前日斜率 }
斜率转正:=短期多空线斜率 > -0.01 AND 前日短期斜率 < -0.01; { 斜率从负转正 }
斜率持续上升:=短期多空线斜率 > 前日短期斜率 AND 前日短期斜率 > 前2日短期斜率; { 连续上升 }

{ 价格条件：收盘价在短期多空线上方 }
价格条件:=CLOSE > 短期多空线;

{ 成交量条件：成交量放大 }
成交量条件:=VOL > MA(VOL, 5) * 1.2;                      { 成交量大于5日均量的1.2倍 }

{ 底部拐点信号定义 }
{ 基础拐点：斜率上升且之前为下降趋势 }
基础拐点:=斜率上升 AND 前日短期斜率 < -0.02;

{ 强化拐点：斜率转正 }
强化拐点:=斜率转正;

{ 完美拐点：多重条件确认 }
完美拐点:=斜率持续上升 AND 价格条件 AND 成交量条件;

{ 底部拐点信号（综合条件） }
拐点信号原始:=基础拐点 AND 强化拐点 AND 完美拐点;

{ 过滤信号：5日内只显示一次 }
底部拐点信号:=FILTER(拐点信号原始,3);

{ 选股条件定义 }
早期买入信号:=CROSS(短期多空线,主力成本) AND 主力斜率条件 AND 机构评分条件;
中期买入信号:=CROSS(短期多空线,长命线) AND 主力斜率条件 AND 机构评分条件;
后期买入信号:=CROSS(短期多空线,强命线)*0.97 AND 主力斜率条件 AND 机构评分条件;
短期拐点信号:=底部拐点信号 AND 主力斜率条件;
机构评分信号:=原始评分信号;

{ 最终选股输出 - 只能有一个输出 }
选股:机构评分信号;
