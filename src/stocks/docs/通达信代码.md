3、B1选股

{ 综合选股公式：基于J值、成交量、趋势强弱等条件筛选 }
{ 公式说明：前2日J值在(13,55)，当日J值<13；量能递减（当日<前2日且昨日<前2日）；白线>黄线；收盘>黄线97% }

{ 参数区}
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=20; SS:=3; { 斜率周期与斜率平滑周期 }
J_LOW:=13; J_HIGH:=60; YELLOW_RATIO:=0.95; { J阈值与黄线比例阈值 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 4) > J_HIGH AND REF(J, 4) > J;       { 前5日J值严格递减，且当日J小于J_LOW }
条件2 := J < J_LOW;                                   { 当日J值小于J_LOW }
条件3 := VOL < REF(VOL, 2) AND VOL < REF(VOL, 1);                        { 当日成交量小于前2日}
条件4 := ZX_WHITE > ZX_YELLOW;                     { 白线大于黄线 }
条件5 := CLOSE > ZX_YELLOW * YELLOW_RATIO;                 { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }
SLOPE_W_RAW := (ZX_WHITE - REF(ZX_WHITE, SP)) / SP;
SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;
SLOPE_W := EMA(SLOPE_W_RAW, SS);
SLOPE_Y := EMA(SLOPE_Y_RAW, SS);
条件6 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 均量递减：MA(VOL,7) > MA(VOL,5) > MA(VOL,3) > VOL }
条件7 := MA(VOL,7) > MA(VOL,5) AND MA(VOL,5) > MA(VOL,3) AND MA(VOL,3) > VOL;

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7;


4、B2选股
{ 综合选股公式：基于J值、涨幅、成交量等多条件筛选 }
{ 公式说明：前日J值<13，当日J值<55，涨幅>3.5%，收阳线，放量，股价在5日均线上方 }

{ 参数区 }
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=30; SS:=3; { 斜率周期与斜率平滑 }
J_LOW:=13; J_HIGH:=55; YELLOW_RATIO:=0.95; { J阈值与黄线比例 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 2) < J_LOW;                          { 前日J值小于J_LOW }
条件2 := J < J_HIGH;                                  { 当日J值小于J_HIGH }

条件4 := CLOSE > OPEN;                            { 当日收盘价大于开盘价 }
条件5 := VOL > REF(VOL, 1);                       { 成交量大于昨日成交量 }
条件6 := CLOSE > MA(CLOSE, 5);                    { 股价（白线）在5日均线（黄线）上方 }
条件7 := ZX_WHITE > ZX_YELLOW;
条件8 := CLOSE > ZX_YELLOW*YELLOW_RATIO;                  { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }

SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;

条件9 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 放量：当日成交量 > 过去5日均量的 1.8 倍 }
条件10 := VOL > 1.8 * MA(VOL, 5);

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7 AND 条件8 AND 条件9 AND 条件10;

{ 公式说明：捕捉缩量回调、技术指标低位、涨幅有限的股票，寻找潜在反弹机会 }

{ 参数区 }
N := 9;    { KDJ周期参数 }
M1 := 3;   { KDJ平滑参数1 }
M2 := 3;   { KDJ平滑参数2 }
天数 := 9;  { 统计天数 }
涨幅阈值 := 5;  { 涨幅阈值（%） }
成交量比例 := 0.25;  { 成交量比例 }

DEN := HHV(HIGH, N) - LLV(LOW, N);
RSV := IF(DEN = 0, 50, (CLOSE - LLV(LOW, N)) / DEN * 100);
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 使用COUNT函数统计7天内涨幅大于5%的交易日数量 }

9、妖股异动
{ 妖股异动选股策略 }
{ 策略说明：捕捉价格接近筹码平均成本、底部长期无量后突然放量、上方套牢盘极少的异动股票 }
{ 适用场景：寻找底部长期横盘后突然启动的潜力股 }

{ ---------- 参数设置 ---------- }
统计周期 := 60;  { 统计周期（日） }
放量倍数 := 2;   { 放量倍数阈值 }
成本偏差阈值 := 0.15;  { 价格与平均成本偏差阈值（15%） }
获利盘阈值 := 0.75;  { 获利盘比例阈值（90%，表示上方套牢盘极少） }
无量比例 := 0.9;  { 无量比例阈值（90%，表示前60日中90%以上为无量状态） }

{ ---------- 条件1：当日收盘价和筹码的平均成本偏差不超过15% ---------- }
{ 筹码平均成本：50%获利盘成本，代表市场平均持仓成本 }
平均成本 := COST(50);
价格偏差 := ABS(CLOSE - 平均成本) / 平均成本;
条件1 := 价格偏差 <= 成本偏差阈值;

{ ---------- 条件2：当日放量，超过前60日成交量均值2倍 ---------- }
{ 计算前60日的平均成交量（排除当日），用于判断当日是否放量 }
前60日均量 := REF(MA(VOL, 统计周期), 1);
条件2 := VOL > 放量倍数 * 前60日均量;

{ ---------- 条件3：前60日一直属于底部无量状态 ---------- }
{ 判断前60日中，成交量相对较小的天数占比 }
{ 使用COUNT统计前60日内成交量小于前60日均量的天数 }
{ COUNT函数统计过去N个周期内满足条件的周期数 }
{ 注意：这里统计的是过去60个周期中，每个周期的成交量小于前60日均量的天数 }
{ 由于前60日均量是固定的（排除当日），所以这个统计是合理的 }
前60日无量天数 := COUNT(VOL < 前60日均量, 统计周期);
条件3基础 := 前60日无量天数 >= 统计周期 * 无量比例;

{ 补充判断：前60日的平均成交量应该较小，确保是底部状态 }
{ 可以进一步判断前60日的最大成交量也相对较小 }
前60日最大量 := REF(HHV(VOL, 统计周期), 1);
底部确认1 := 前60日最大量 < 前60日均量 * 3;  { 前60日最大量不超过均量的2.5倍 }
条件3增强 := 条件3基础 AND 底部确认1;

{ ---------- 条件4：当前成交价上方基本无套牢盘 ---------- }
{ 获利盘比例：当前价位上方的筹码占比 }
{ WINNER(CLOSE) 返回当前收盘价上方的获利盘比例 }
当前获利盘 := WINNER(CLOSE);
条件4 := 当前获利盘 >= 获利盘阈值;

{ 补充判断：上方套牢盘极少，可以用90%和95%获利盘成本来判断 }
{ 如果当前价格接近或高于90%获利盘成本，说明上方套牢盘极少 }
上方成本90 := COST(90);
上方成本95 := COST(95);
套牢盘判断 := CLOSE >= 上方成本90 * 0.98;  { 收盘价接近或高于90%获利盘成本 }
条件4增强 := 条件4 OR 套牢盘判断;

{ ---------- 综合选股条件 ---------- }
妖股异动信号 := 条件1 AND 条件2 AND 条件3增强 AND 条件4;

{ 输出选股结果 }
选股: 妖股异动信号;

{ 策略说明：筛选近期出现涨停的个股，在涨停后成交量快速萎缩，确认缩量回踩的交易机会 }
{ 适用场景：寻找涨停后回调缩量、筹码保持稳定的股票 }

{ ---------- 参数设置 ---------- }
统计天数B := 10;      { 统计周期（日） }
涨停阈值 := 9.5;      { 涨停阈值（%） }
最大涨停次数 := 3;    { 10日内涨停次数上限（小于3次） }

{ 每日涨幅 = (当日收盘价 - 前一日收盘价) / 前一日收盘价 * 100 }
每日涨幅 := (CLOSE - REF(CLOSE, 1)) / REF(CLOSE, 1) * 100;
涨停条件 := 每日涨幅 >= 涨停阈值;
涨停次数 := COUNT(涨停条件, 统计天数B);

首次涨停量 := IF(涨停次数 > 0, VALUEWHEN(涨停条件, VOL), 0);
条件2B := 涨停次数 > 0 AND 首次涨停量 > 0 AND VOL < 首次涨停量 * 成交量比例B;

11、分歧一致选股
{ 量能递增回调选股策略 }
{ 策略说明：前3日量能逐步增大，前3日连续小幅度上涨，今日跌幅在-1%到0%之间，且比前一日成交量小，当日成交量大于前第4日2倍 }

{ ---------- 参数设置 ---------- }
小涨阈值 := 3;  { 小幅度上涨阈值（%） }
跌幅上限 := 1;  { 跌幅上限（%） }
跌幅下限 := -2; { 跌幅下限（%） }

{ ---------- 条件1：前3日量能逐步增大 ---------- }
量能逐步增大 := REF(VOL, 1) > REF(VOL, 2) AND REF(VOL, 2) > REF(VOL, 3);

{ ---------- 条件2：前3日连续小幅度上涨 ---------- }
{ 计算每日涨幅：(当日收盘 - 前日收盘) / 前日收盘 * 100 }
前1日涨幅 := (REF(CLOSE, 1) - REF(CLOSE, 2)) / REF(CLOSE, 2) * 100;
前2日涨幅 := (REF(CLOSE, 2) - REF(CLOSE, 3)) / REF(CLOSE, 3) * 100;
前3日涨幅 := (REF(CLOSE, 3) - REF(CLOSE, 4)) / REF(CLOSE, 4) * 100;

{ 判断前3日涨幅都在0到小涨阈值之间 }
连续小涨 := 前1日涨幅 > 0 AND 前1日涨幅 < 小涨阈值 AND 
            前2日涨幅 > 0 AND 前2日涨幅 < 小涨阈值 AND 
            前3日涨幅 > 0 AND 前3日涨幅 < 小涨阈值;

{ ---------- 条件3：今日跌幅在-1%到0%之间 ---------- }
当日跌幅 := (CLOSE - REF(CLOSE, 1)) / REF(CLOSE, 1) * 100;
跌幅符合 := 当日跌幅 >= 跌幅下限 AND 当日跌幅 <= 跌幅上限;

{ ---------- 条件4：今日比前一日成交量小 ---------- }
今日缩量 := VOL < REF(VOL, 1);

{ ---------- 条件5：当日成交量大于前第4日2倍（有效放量） ---------- }
有效放量 := VOL > REF(VOL, 4) * 2;

{ ---------- 综合选股条件 ---------- }
选股: 量能逐步增大 AND 连续小涨 AND 跌幅符合 AND 今日缩量 AND 有效放量;
