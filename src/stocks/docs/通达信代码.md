3、B1选股

{ 综合选股公式：基于J值、成交量、趋势强弱等条件筛选 }
{ 公式说明：前2日J值在(13,55)，当日J值<13；量能递减（当日<前2日且昨日<前2日）；白线>黄线；收盘>黄线97% }

{ 参数区}
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=20; SS:=3; { 斜率周期与斜率平滑周期 }
J_LOW:=13; J_HIGH:=60; YELLOW_RATIO:=0.95; { J阈值与黄线比例阈值 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 4) > J_HIGH AND REF(J, 4) > J;       { 前5日J值严格递减，且当日J小于J_LOW }
条件2 := J < J_LOW;                                   { 当日J值小于J_LOW }
条件3 := VOL < REF(VOL, 2) AND VOL < REF(VOL, 1);                        { 当日成交量小于前2日}
条件4 := ZX_WHITE > ZX_YELLOW;                     { 白线大于黄线 }
条件5 := CLOSE > ZX_YELLOW * YELLOW_RATIO;                 { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }
SLOPE_W_RAW := (ZX_WHITE - REF(ZX_WHITE, SP)) / SP;
SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;
SLOPE_W := EMA(SLOPE_W_RAW, SS);
SLOPE_Y := EMA(SLOPE_Y_RAW, SS);
条件6 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 均量递减：MA(VOL,7) > MA(VOL,5) > MA(VOL,3) > VOL }
条件7 := MA(VOL,7) > MA(VOL,5) AND MA(VOL,5) > MA(VOL,3) AND MA(VOL,3) > VOL;

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7;


4、B2选股
{ 综合选股公式：基于J值、涨幅、成交量等多条件筛选 }
{ 公式说明：前日J值<13，当日J值<55，涨幅>3.5%，收阳线，放量，股价在5日均线上方 }

{ 参数区 }
P1:=5; P2:=10; P3:=20; P4:=60;
SP:=30; SS:=3; { 斜率周期与斜率平滑 }
J_LOW:=13; J_HIGH:=55; YELLOW_RATIO:=0.95; { J阈值与黄线比例 }
TOP_N:=20; W_UP:=0.03; W_LO:=0.03; { 最近山顶窗口与上/下影阈值 }

{ 知行趋势线 }
ZX_WHITE:=EMA(EMA(C,10),10);
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 定义选股条件 }
条件1 := REF(J, 2) < J_LOW;                          { 前日J值小于J_LOW }
条件2 := J < J_HIGH;                                  { 当日J值小于J_HIGH }
条件3 := (CLOSE - REF(CLOSE, 1)) / REF(CLOSE, 1) * 100 > 3.95; { 当日涨幅大于3.95% }
条件4 := CLOSE > OPEN;                            { 当日收盘价大于开盘价 }
条件5 := VOL > REF(VOL, 1);                       { 成交量大于昨日成交量 }
条件6 := CLOSE > MA(CLOSE, 5);                    { 股价（白线）在5日均线（黄线）上方 }
条件7 := ZX_WHITE > ZX_YELLOW;
条件8 := CLOSE > ZX_YELLOW*YELLOW_RATIO;                  { 收盘价高于黄线比例 }

{ 趋势斜率（参数化+平滑）：以 SP 日差分/SP，并用 SS EMA 平滑 }
SLOPE_W_RAW := (ZX_WHITE - REF(ZX_WHITE, SP)) / SP;
SLOPE_Y_RAW := (ZX_YELLOW - REF(ZX_YELLOW, SP)) / SP;
SLOPE_W := EMA(SLOPE_W_RAW, SS);
SLOPE_Y := EMA(SLOPE_Y_RAW, SS);
条件9 := SLOPE_W > 0 AND SLOPE_Y > 0;              { 白/黄线 SP 日斜率向上（SS 平滑） }

{ 放量：当日成交量 > 过去5日均量的 1.8 倍 }
条件10 := VOL > 1.8 * MA(VOL, 5);

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7 AND 条件8 AND 条件9 AND 条件10;
