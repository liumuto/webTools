1、我的策略选股（牛市策略1：已经突破）
{ 综合选股公式：基于J值、成交量、趋势强弱等条件筛选 }

{ 参数区}
P1:=14; P2:=28; P3:=57; P4:=114;

{ 知行短期趋势线（白线） }
ZX_WHITE:=EMA(EMA(C,10),10);

{ 知行多空线（黄线） }
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{BBI线}
BBI:= (MA(C,3)+MA(C,6)+MA(C,12)+MA(C,24))/4;

{ 计算KDJ指标中的J值 }
N := 9;    { KDJ周期参数 }
M1 := 3;
M2 := 3;
RSV := (CLOSE - LLV(LOW, N)) / (HHV(HIGH, N) - LLV(LOW, N)) * 100;
K := SMA(RSV, M1, 1);
D := SMA(K, M2, 1);
J := 3 * K - 2 * D;

{ 计算KDJ-K和KDJ-D的3日线性拟合斜率 }
KDJ_K斜率 := SLOPE(K, 3);                          { KDJ-K近3日线性拟合斜率 }
KDJ_D斜率 := SLOPE(D, 3);                          { KDJ-D近3日线性拟合斜率 }

{ 计算BBI和白线的3日线性拟合斜率 }
BBI斜率5日 := SLOPE(BBI, 5);                          { BBI近5日线性拟合斜率 }
BBI斜率3日 := SLOPE(BBI, 3);                          { BBI近3日线性拟合斜率 }
白线斜率 := SLOPE(ZX_WHITE, 5);                    { 白线近5日线性拟合斜率 }
黄线斜率 := SLOPE(ZX_YELLOW, 45);                    { 黄线近45日线性拟合斜率 }

{ 成交量条件 }
条件7 := VOL > MA(VOL, 5) * 1.2;;                    { 当日成交量大于前5天日均成交量1.2倍 }


{ 定义选股条件 }
条件1 := KDJ_K斜率 > 0;                    { KDJ-K斜率大于0 }
条件2 := KDJ_K斜率 > KDJ_D斜率;                    { KDJ-K斜率大于KDJ-D斜率 }
条件3 := BBI > ZX_WHITE AND REF(BBI, 1) < REF(ZX_WHITE, 1);  { 当日BBI大于等于白线且前1日BBI小于白线 }
条件4 := BBI斜率3日 > 白线斜率;                        { BBI斜率大于白线斜率 }
条件5 := BBI斜率5日 > 0 AND BBI斜率5日 <= 0.3 AND BBI斜率3日 <= 0.1;          { BBI斜率5日<0.3且3日<0.1 }
条件6 := 黄线斜率 > 0;                        { 黄线斜率大于0 }
条件8 := ZX_WHITE > ZX_YELLOW;                { 白线值大于黄线值 }

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND 条件5 AND 条件6 AND 条件7 AND 条件8;
