1、趋势策略选股（牛市策略：即将突破）
{ 综合选股公式：基于J值、成交量、趋势强弱等条件筛选 }

{ 参数区}
P1:=14; P2:=28; P3:=57; P4:=114;

{ 知行短期趋势线（白线） }
ZX_WHITE:=EMA(EMA(C,10),10);

{ 知行多空线（黄线） }
ZX_YELLOW:=(MA(C,P1)+MA(C,P2)+MA(C,P3)+MA(C,P4))/4;

{BBI线}
BBI:= (MA(C,3)+MA(C,6)+MA(C,12)+MA(C,24))/4;

{ 计算BBI和白线的3日线性拟合斜率 }
BBI斜率3日 := SLOPE(BBI, 3);                          { BBI近3日线性拟合斜率 }
黄线斜率 := SLOPE(ZX_YELLOW, 60);                    { 黄线近60日线性拟合斜率 }
白线斜率3日 := SLOPE(ZX_WHITE, 3);                    { 白线近3日线性拟合斜率 }
均值_当日 := (ZX_WHITE + ZX_YELLOW + BBI) / 3;        { 三者当日均值 }
标准差_当日 := SQRT((POW(ZX_WHITE-均值_当日,2) + POW(ZX_YELLOW-均值_当日,2) + POW(BBI-均值_当日,2)) / 3); { 三者当日标准差 }
均值_当日_2 := (ZX_WHITE + BBI) / 2;                 { 白线与BBI两者当日均值 }
标准差_当日_2 := SQRT((POW(ZX_WHITE-均值_当日_2,2) + POW(BBI-均值_当日_2,2)) / 2); { 白线与BBI两者当日标准差 }


{ 定义选股条件 }
条件1 := 黄线斜率 > 0.005;                        { 黄线斜率大于0.005 }
条件2 := ZX_WHITE >= ZX_YELLOW;                { 白线值大于黄线值 }

{ 市盈率条件 }
条件3 := DYNAINFO(39) > 0;                           { 股票市盈率大于0 }

{ 新增条件：白线/BBI 近3日斜率关系 }
条件4 := ABS(白线斜率3日) <= 0.015;                    { 白线近3日斜率绝对值小于0.015 }
条件5:= 白线斜率3日 != 0 AND BBI斜率3日 > 0 AND      
          (BBI斜率3日 - 白线斜率3日) / ABS(白线斜率3日) > 3;  { 白线3日斜率非0时的比例判断 }
条件6 := 白线斜率3日 = 0 AND BBI斜率3日 >= 0.03;       { 白线3日斜率为0时，BBI3日斜率>=0.03 }
条件7 := 标准差_当日 <= 0.25;                        { 当日白/黄/BBI三者标准差≤0.25 }
条件8 := 标准差_当日_2 <= 0.03;                      { 当日白线与BBI两者标准差≤0.03 }

{ 综合选股条件 }
选股: 条件1 AND 条件2 AND 条件3 AND 条件4 AND (条件5 OR 条件6) AND 条件7 AND 条件8;

3、底部放量异动

{ 参数设置（可按需调整） }
N:=20;            { 横盘/成交量参考周期，建议20~60 }
M:=0.2;          { 振幅阈值，例如20% }
ATRN:=20;         { ATR计算周期，建议与N接近 }
K:=1.6;           { 放量倍数，例如1.6倍 }
STD_TH:=0.20;     { 均线粘合的STD阈值，可在0.10~0.20间微调 }
RISE_TH:=0.06;    { 可选：当日涨幅阈值6% }
OVER_UP:=1.08;    { 突破后不远离前高，当前价/前高 < 1.08 }
TURN_MIN:=3;      { 换手率下限1% }
TURN_MAX:=15;     { 换手率上限15% }

{ 基础量价与波动度计算 }
TRUE_RANGE:=MAX(MAX(H-L,ABS(H-REF(C,1))),ABS(L-REF(C,1))); { 避免与内置TR函数名冲突 }
ATR_VAL:=MA(TRUE_RANGE,ATRN);
横盘_振幅:= (HHV(C,N)-LLV(C,N))/MAX(LLV(C,N),0.01); { 保护分母 }
横盘_振幅条件:= 横盘_振幅 <= M;
横盘_ATR条件:= ATR_VAL/MA(C,N) < 0.03;
横盘_均线粘合:= STD(MA(C,5),10) < STD_TH;
横盘判定:= 横盘_振幅条件 OR 横盘_ATR条件 OR 横盘_均线粘合;

{ 放量与价格突破条件 }
放量倍数:= VOL/MA(VOL,N);
放量条件:= 放量倍数 > K;
上轨:= HHV(H,N);
价格突破:= C > 上轨 AND C > O;
可选涨幅:= (C-REF(C,1))/MAX(REF(C,1),0.01) > RISE_TH;

{ 排除无效突破 }
大阴线:= C < O AND (O-C)/MAX(REF(C,1),0.01) > 0.03;
连续大阴:= 大阴线 AND REF(大阴线,1);
无连续大阴:= COUNT(连续大阴,5) = 0;
不过度远离前高:= C/REF(上轨,1) < OVER_UP;
{ 简化换手率计算：以流通盘估算，通达信中CAPITAL为流通股(万股) }
换手率:= IF(CAPITAL>0, VOL/CAPITAL*100, 0);
换手合理:= 换手率 > TURN_MIN AND 换手率 < TURN_MAX;

{ 综合信号（默认包含“可选涨幅”，如需关闭可将其改为TRUE） }
信号触发:= 横盘判定 AND 放量条件 AND 价格突破 AND 可选涨幅 AND 无连续大阴 AND 不过度远离前高 AND 换手合理;

{ 选股输出与排序参考指标（条件选股仅允许单一输出） }
XG: 信号触发;
放量参考:= 放量倍数;  { 内部变量，非输出 }
突破力度:= C/上轨;    { 内部变量，非输出 }
